const inputs = `23krgjlpone
kfxone67bzb2
8jjpseven
236twoknbxlczgd
...`
let stringArray = inputs.split('\n');
const values = {
  one: 1,
  two: 2,
  three: 3,
  four: 4,
  five: 5,
  six: 6,
  seven: 7,
  eight: 8,
  nine: 9
}

// Convert string numbers to numbers 
stringArray = stringArray.map(el => {
  let newEl = el;
  Object.keys(values).map(value => {
      newEl = newEl.replaceAll(`${value}`, `${values[value]}`);
    })     
    return newEl
})
let numStrArray = stringArray.map(str => str.replace(/[^0-9]/g, ''));
numStrArray = numStrArray.map(el => el[0] + el[el.length-1])
numStrArray = numStrArray.map(str => parseInt(str));
const answer = numStrArray.reduce((acc, next) => acc + parseInt(next));
alert('answer:', answer);
